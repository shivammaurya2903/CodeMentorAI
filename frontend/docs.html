Code Review Guide

Example Review

// Original code
function findDuplicates(arr) {
  const result = [];
  for (let i = 0; i < arr.length; i++) {
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[i] === arr[j]) {
        result.push(arr[i]);
      }
    }
  }
  return result;
}

AI Feedback: O(n²) complexity detected. Here's an optimized O(n) solution using hashmap:

// Optimized version ✅ Score: 95/100
function findDuplicates(arr) {
  const hash = {};
  const duplicates = [];
  
  for (const num of arr) {
    if (hash[num]) {
      duplicates.push(num);
    } else {
      hash[num] = true;
    }
  }
  return [...new Set(duplicates)];
}
